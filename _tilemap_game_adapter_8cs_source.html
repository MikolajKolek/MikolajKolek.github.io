147 
148  if (tilemap.GetTile(newPlayerLocation) == wall || tilemap.GetTile(newPlayerLocation) == empty || tilemap.GetTile(newPlayerLocation) == null)
149  return false;
150  if (tilemap.GetTile(newPlayerLocation) == box || tilemap.GetTile(newPlayerLocation) == boxOnBoxArea)
151  {
152  if (tilemap.GetTile(newPlayerLocation + direction) == wall || tilemap.GetTile(newPlayerLocation + direction) == box || tilemap.GetTile(newPlayerLocation + direction) == boxOnBoxArea || tilemap.GetTile(newPlayerLocation + direction) == empty || tilemap.GetTile(newPlayerLocation + direction) == null)
153  return false;
154

156  moveClip = AudioManager.AudioEffectClip.BoxMovedInPlace;
157  boxInPlaceCount++;
158  boxPlacedOnBoxArea = true;
159  tilemap.SetTile(newPlayerLocation + direction, boxOnBoxArea);
160  }
161  if (tilemap.GetTile(newPlayerLocation) == boxOnBoxArea)
162  boxInPlaceCount--;

164  moveClip = AudioManager.AudioEffectClip.LevelFinished;

202  public void WindowScaleUpdate(int levelHeight, int levelWidth) {
203  //The tileAnchor is an offset at which the tile with the position x=0, y=0 is located on the screen.
204  //By default the tileAnchor is always set to 0, 0, 0, which means the tile at position x=0, y=0
205  //is located at the exact center of the screen.
206  //If we were to change the tileAnchor's value to 1, 1, 0 for example, the location of the tile at the exact center of the screen
207  //would change to x=-1, y=-1 and the center of the tilemap would be offset by exactly 1 tile to the top and one tile to the right.
208  var baseTileAnchor = new Vector3(0, 0, 0);
209 
210  //The currentTileSideLength variable is set to the current length of a tile's side. Tiles are always square by default, and
211  //although I couldn't find any documentation stating this, I found that by default in all possible screen orientations
212  //a Tilemap fits exactly 10 tiles vertically on the screen.
213  var currentTileSideLength = Screen.height / 10f;
214  //The horizontalTileCount variable stores the information of how many tiles currently fit on the screen in the area set by
215  //horizontalTilemapSpan excluding the horizontal margins.
216  var horizontalTileCount = (Screen.width - horizontalMargin * 2) / currentTileSideLength * horizontalTilemapSpan;
217  //The horizontalScale is a scale value that if the scale was set to, exactly levelWidth tiles would fit on the area set by
218  //horizontalTilemapSpan and horizontalMargin
219  var horizontalScale = horizontalTileCount / levelWidth;
220  //The verticalTileCount variable stores information on exactly how many tiles can fit vertically on the screen excluding the
221  //vertical margins
222  var verticalTileCount = 10f - ((10f / Screen.height) * 2 * verticalMargin);
223  //The verticalScale is a scale value that if the scale was set to, exactly levelHeight tiles would fit on the area set by
224  //verticalMarginMargin
225  var verticalScale = verticalTileCount / levelHeight;
226  //The final scale value selects the lower one of horizontalScale and verticalScale. This way, at all times, at least
227  //levelHeight tiles will fit on the screen vertically, and at least levelWidth tiles will fit on the screen horizontally
228  var scale = Math.Min(horizontalScale, verticalScale);
229 
230  //The newTileSideLength variable stores the information on how long the sides of a tile will be with the new scale value.
231  var newTileSideLength = currentTileSideLength * scale;
232  tileSideLength = newTileSideLength;
233 
234  //The baseTileAnchor is offset to the left the exact number that is needed for the x=0 column to be at the left edge of the area
235  //the borders of which are set by horizontalTilemapSpan and horizontalMargin
236  baseTileAnchor.x -= (Screen.width / newTileSideLength) * horizontalMultiplier - 0.5f;
237  //The baseTileAnchor is offset to the left the exact number that is needed for it to be located at the exact centre of the area
238  //the borders of which are set by horizontalTilemapSpan and horizontalMargin
239  baseTileAnchor.x += (((Screen.width / newTileSideLength) * horizontalTilemapSpan) - levelWidth) * 0.5f;
240  //The baseTileAnchor is offset to the left by the exact number that is needed for it to be located at the exact vertical center
241  //of the screen.
242  //The 0.5f needs to be subtracted because when the tileAnchor's y coordinate is dividable by 1, an odd number of tiles can
243  //be displayed on the screen vertically, while when the tileAnchor's y coordinate isn't dividable by 1 but is dividable by 0,5
244  //an even number of tiles can be displayed on the screen vertically.
245  baseTileAnchor.y -= (0.5f * levelHeight) - 0.5f;
246 
247  //The baseXCoordinate variable is set to the exact x coordinate of the left corner of the tile that with the tileAnchor.x = 0
248  //would be located at the center of the screen
249  baseXCoordinate = (Screen.width / 2f) - (newTileSideLength / 2f);
250  //The baseXCoordinate is offset in such a way that it is now set to the exact x coordinate of the left corner of the tile that will
251  //be located at x = 0 on the tilemap when the baseTilemapAnchor becomes the new tilemapAnchor
252  baseXCoordinate += newTileSideLength * baseTileAnchor.x;
253  //The baseYCoordinate variable is set to the exact y coordinate of the bottom corner of the tile that with the tileAnchor.y = 0
254  //would be located at the center of the screen
255  baseYCoordinate = (Screen.height / 2f) - (newTileSideLength / 2f);
256  //The baseYCoordinate is offset in such a way that it is now set to the exact y coordinate of the bottom corner of the tile that will
257  //be located at y = 0 on the tilemap when the baseTilemapAnchor becomes the new tilemapAnchor
258  baseYCoordinate += newTileSideLength * baseTileAnchor.y;
259 
260  //The actual tilemap's tileAnchor is set to the baseTileAnchor.
261  tilemap.tileAnchor = baseTileAnchor;
262 
263  //And the scale is set to the new scale.
264  var vectorScale = new Vector3(scale, scale);
265  grid.transform.localScale = vectorScale;
266  }

273  public void UpdateLevelSize(int levelHeight, int levelWidth) {
274  //The horizontalMultiplier is set to where the left border of the tilemap will be with the current horizontalMultiplier
275  //(but excluding the current horizontalMargin). It is stored as a float, and for example:
276  //0f would mean the left border is at the middle of the screen
277  //0.25f would mean that the border is a quarter of the screen offset from the center of the screen to the right, so it fills 75% of the screen.
278 
279  //For example, with horizontalTilemapSpan = 0.25f the tilemap will only take up the right 25% of the screen, meaning that
280  //the horizontalMultiplier will be equal to -0.25f.
281  if (horizontalTilemapSpan >= 0.5f)
282  horizontalMultiplier = horizontalTilemapSpan - 0.5f;
283  else
284  horizontalMultiplier = horizontalTilemapSpan - 0.5f;
285 
286  WindowScaleUpdate(levelHeight, levelWidth);
287  }
